import React, { useState } from 'react';

const DOK1Assessment = ({ document }) => {
  const [activeTab, setActiveTab] = useState('grading');
  
  // Classification types
  const CLASSIFICATION = {
    BRAINLIFT: 'brainlift',
    PARTIAL: 'partial',
    NOT_BRAINLIFT: 'not_brainlift'
  };
  
  // Example document data structure
  const defaultDocument = {
    title: "Zach Groshell - Direct Instruction",
    classification: CLASSIFICATION.NOT_BRAINLIFT,
    subtype: "Compound Doc (Design Manual + Project Doc + Reference List)",
    gradeableFacts: 0,
    reason: "Document contains 30+ academic citations but zero statements of what those papers found. All content is prescriptive (\"You should...\") or operational (rubrics, specs, project scope).",
    recommendation: "Convert to brainlift by adding factual claims like \"Wouters & van Oostendorp (2013) meta-analysis found games with instructional support produced effect size of X vs. Y for games alone.\"",
    // For actual brainlifts, these would be populated:
    facts: [],
    contradictions: [],
    readingList: []
  };
  
  const doc = document || defaultDocument;
  
  // Classification badge component
  const ClassificationBadge = ({ classification }) => {
    const badges = {
      [CLASSIFICATION.BRAINLIFT]: {
        icon: '✅',
        text: 'BRAINLIFT',
        bgColor: 'bg-green-100',
        textColor: 'text-green-800',
        borderColor: 'border-green-300'
      },
      [CLASSIFICATION.PARTIAL]: {
        icon: '⚠️',
        text: 'PARTIAL BRAINLIFT',
        bgColor: 'bg-yellow-100',
        textColor: 'text-yellow-800',
        borderColor: 'border-yellow-300'
      },
      [CLASSIFICATION.NOT_BRAINLIFT]: {
        icon: '❌',
        text: 'NOT A BRAINLIFT',
        bgColor: 'bg-red-100',
        textColor: 'text-red-800',
        borderColor: 'border-red-300'
      }
    };
    
    const badge = badges[classification];
    
    return (
      <span className={`inline-flex items-center px-3 py-1 rounded-full text-sm font-medium ${badge.bgColor} ${badge.textColor} border ${badge.borderColor}`}>
        <span className="mr-1">{badge.icon}</span>
        {badge.text}
      </span>
    );
  };
  
  // Not a Brainlift view
  const NotBrainliftView = () => (
    <div className="bg-gray-50 rounded-lg p-6 mt-4">
      <div className="text-center mb-6">
        <div className="text-6xl mb-4">📄</div>
        <h2 className="text-xl font-semibold text-gray-700">
          This document cannot be graded under DOK1 methodology
        </h2>
      </div>
      
      <div className="max-w-2xl mx-auto space-y-4">
        <div className="bg-white rounded-lg p-4 border border-gray-200">
          <div className="text-sm text-gray-500 uppercase tracking-wide mb-1">Subtype</div>
          <div className="text-gray-900 font-medium">{doc.subtype}</div>
        </div>
        
        <div className="bg-white rounded-lg p-4 border border-gray-200">
          <div className="text-sm text-gray-500 uppercase tracking-wide mb-1">Gradeable Facts</div>
          <div className="text-3xl font-bold text-red-600">{doc.gradeableFacts}</div>
        </div>
        
        <div className="bg-white rounded-lg p-4 border border-gray-200">
          <div className="text-sm text-gray-500 uppercase tracking-wide mb-1">Reason</div>
          <div className="text-gray-700">{doc.reason}</div>
        </div>
        
        <div className="bg-blue-50 rounded-lg p-4 border border-blue-200">
          <div className="text-sm text-blue-600 uppercase tracking-wide mb-1">To Make Gradeable</div>
          <div className="text-blue-900">{doc.recommendation}</div>
        </div>
      </div>
      
      <div className="flex justify-center gap-3 mt-6">
        <button className="px-4 py-2 bg-gray-800 text-white rounded-lg hover:bg-gray-700 transition-colors">
          Download Report
        </button>
        <button className="px-4 py-2 bg-white text-gray-800 border border-gray-300 rounded-lg hover:bg-gray-50 transition-colors">
          Share
        </button>
      </div>
    </div>
  );
  
  // Tabs for gradeable documents
  const GradeableTabs = () => (
    <div className="mt-4">
      <div className="border-b border-gray-200">
        <nav className="flex space-x-8">
          {['grading', 'contradictions', 'readingList'].map((tab) => (
            <button
              key={tab}
              onClick={() => setActiveTab(tab)}
              className={`py-4 px-1 border-b-2 font-medium text-sm transition-colors ${
                activeTab === tab
                  ? 'border-blue-500 text-blue-600'
                  : 'border-transparent text-gray-500 hover:text-gray-700 hover:border-gray-300'
              }`}
            >
              {tab === 'grading' && 'Fact Grading'}
              {tab === 'contradictions' && 'Contradictions'}
              {tab === 'readingList' && 'Reading List'}
            </button>
          ))}
        </nav>
      </div>
      
      <div className="py-6">
        {activeTab === 'grading' && (
          <div className="text-gray-500 text-center py-12">
            Fact grading table would render here
          </div>
        )}
        {activeTab === 'contradictions' && (
          <div className="text-gray-500 text-center py-12">
            Contradiction clusters would render here
          </div>
        )}
        {activeTab === 'readingList' && (
          <div className="text-gray-500 text-center py-12">
            Reading list would render here
          </div>
        )}
      </div>
    </div>
  );
  
  // Partial brainlift warning banner
  const PartialWarning = () => (
    <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-4 mt-4 flex items-start">
      <span className="text-yellow-500 mr-3 text-xl">⚠️</span>
      <div>
        <div className="font-medium text-yellow-800">Partial Brainlift</div>
        <div className="text-yellow-700 text-sm">
          Only {doc.gradeableFacts} of {doc.totalClaims || '?'} claims in this document are gradeable. 
          Non-gradeable content (prescriptive statements, uncited claims) has been excluded.
        </div>
      </div>
    </div>
  );
  
  return (
    <div className="max-w-4xl mx-auto p-6">
      {/* Header */}
      <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
        <div className="flex items-start justify-between">
          <div>
            <div className="text-sm text-gray-500 uppercase tracking-wide mb-1">DOK1 Assessment</div>
            <h1 className="text-2xl font-bold text-gray-900">{doc.title}</h1>
          </div>
          <ClassificationBadge classification={doc.classification} />
        </div>
        
        {/* Summary stats for gradeable docs */}
        {doc.classification !== CLASSIFICATION.NOT_BRAINLIFT && (
          <div className="grid grid-cols-4 gap-4 mt-6 pt-6 border-t border-gray-100">
            <div>
              <div className="text-sm text-gray-500">Total Facts</div>
              <div className="text-2xl font-bold text-gray-900">{doc.facts?.length || 0}</div>
            </div>
            <div>
              <div className="text-sm text-gray-500">Mean Score</div>
              <div className="text-2xl font-bold text-gray-900">
                {doc.facts?.length ? (doc.facts.reduce((a, f) => a + f.score, 0) / doc.facts.length).toFixed(2) : '-'}
              </div>
            </div>
            <div>
              <div className="text-sm text-gray-500">Score 5 (Verified)</div>
              <div className="text-2xl font-bold text-green-600">
                {doc.facts?.filter(f => f.score === 5).length || 0}
              </div>
            </div>
            <div>
              <div className="text-sm text-gray-500">Contradictions</div>
              <div className="text-2xl font-bold text-orange-600">
                {doc.contradictions?.length || 0}
              </div>
            </div>
          </div>
        )}
      </div>
      
      {/* Conditional content based on classification */}
      {doc.classification === CLASSIFICATION.NOT_BRAINLIFT && <NotBrainliftView />}
      
      {doc.classification === CLASSIFICATION.PARTIAL && (
        <>
          <PartialWarning />
          <GradeableTabs />
        </>
      )}
      
      {doc.classification === CLASSIFICATION.BRAINLIFT && <GradeableTabs />}
    </div>
  );
};
